namespace Neuron\Data\Validation;

/**
 * Class IPAddress
 * @package Neuron\Data\Validation
 */

class IPAddress
	extends Validator
{
	protected function validate( $s )
	{
		return filter_var( $s, FILTER_VALIDATE_IP );
	}
}

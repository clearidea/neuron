namespace Neuron\Data\Validation;

/**
 * Class Integer
 * @package Neuron\Data\Validation
 */

class Integer
	extends Validator
{
	protected function validate( $i )
	{
		if( is_int( $i ) )
			return true;

		return ctype_digit( $i );
	}
}
